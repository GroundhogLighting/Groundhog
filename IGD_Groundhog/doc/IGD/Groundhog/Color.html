def self.colortrans(c,mat)
    cout = []
	cout[0] = mat[0][0]*c[0] + mat[0][1]*c[1] + mat[0][2]*c[2];
	cout[1] = mat[1][0]*c[0] + mat[1][1]*c[1] + mat[1][2]*c[2];
	cout[2] = mat[2][0]*c[0] + mat[2][1]*c[1] + mat[2][2]*c[2];
    return cout
end
